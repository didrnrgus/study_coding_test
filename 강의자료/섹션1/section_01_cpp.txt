std::pair<int, int> solution(const std::vector<int>& arr)
{

	int oddVal = INT_MAX;

	for (int a : arr)
	{
		if (a % 2 == 1)
		{
			result.first += a;
			oddVal = std::min(oddVal, a);
		}
	}

	result.second = oddVal;
	return result;
}
