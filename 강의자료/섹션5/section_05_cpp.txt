int solution(const std::string& s, const std::string& t)
{
    std::unordered_map<char, int> tMap, sMap;

    // t ë¬¸ìì—´ì˜ ë¬¸ì ê°œìˆ˜ë¥¼ í•´ì‹œë§µì— ì €ì¥
    for (char c : t) tMap[c]++;

    int len = static_cast<int>(t.size());

    // ìŠ¬ë¼ì´ë”© ìœˆë„ìš° ì¤€ë¹„: ì²˜ìŒ len - 1ê¹Œì§€ ë¬¸ì ì¹´ìš´íŒ…
    for (int i = 0; i < len - 1; i++)
        sMap[s[i]]++;

    int lt = 0, answer = 0;

    // ìŠ¬ë¼ì´ë”© ìœˆë„ìš° ì‹œì‘
    for (int rt = len - 1; rt < static_cast<int>(s.size()); rt++)
    {
        sMap[s[rt]]++; // ì˜¤ë¥¸ìª½ ë¬¸ì ì¶”ê°€

        // í˜„ì¬ ìœˆë„ìš° ë‚´ ë¬¸ì êµ¬ì„±ê³¼ tMap ë¹„êµ
        if (compareMaps(sMap, tMap)) 
            answer++;

        // ì™¼ìª½ ë¬¸ì ì œê±° (ìœˆë„ìš° ì´ë™ ì¤€ë¹„)
        sMap[s[lt]]--;
        if (sMap[s[lt]] == 0) 
            sMap.erase(s[lt]); // ê°œìˆ˜ê°€ 0ì´ë©´ ë§µì—ì„œ ì œê±°

        lt++; // ì™¼ìª½ í¬ì¸í„° ì´ë™
    }

    return answer;
}
